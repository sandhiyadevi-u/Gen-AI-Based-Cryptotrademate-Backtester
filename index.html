Quick Start Example

from cryptotrademate import BacktestEngine
from strategies import MovingAverageCrossover

# Initialize backtesting engine
engine = BacktestEngine('BTC-USD', '2020-01-01', '2024-01-01')

# Create and run strategy
strategy = MovingAverageCrossover(fast_period=50, slow_period=200)
results = engine.run_backtest(strategy)

# Analyze results
print(f"Total Return: {results.total_return:.2f}%")
print(f"Sharpe Ratio: {results.sharpe_ratio:.3f}")
print(f"Max Drawdown: {results.max_drawdown:.2f}%")

# Generate comprehensive report
results.generate_report('backtest_results.html')
